```
class Circle():
    def __init__(self, x, y=0):
        self.x = x
        self.y = y

    def center(self, a=0, b=0):
        return (a,b)

    def move(self, a, b):
        self.a = a
        self.b = b
        return (a, b)

c1 = Circle(3)
print(c1.center())
print(c1.move())
```
